goal: (function | statement)*

function: "def" WORD "(" argument? ")" ":" statement*

argument: WORD ("=" SIGNED_NUMBER )? ( "," WORD ("=" SIGNED_NUMBER )?)*

statement: "if" comparison ":" statement*                                    -> if
    | "while" comparison ":" statement*                                      -> while
    | "for" WORD "in" WORD ":" statement*                                    -> for
    | "return" sum                                                           -> return
    | WORD "=" sum                                                           -> variable_assignment
    | "var" WORD "=" sum                                                     -> variable_creation
    | WORD "[" sum "]" "=" sum                                               -> array_assignment
    | function_call                                                          -> call

disjunction: conjunction ("or" conjunction)+                                -> or
    | conjunction                                                           -> conjunction

conjunction: comparison ("and" comparison)+                                 -> and
    | comparison                                                            -> comparison

comparison: sum "<=" sum                                                    -> less_equals
    | sum "<" sum                                                           -> less
    | sum ">=" sum                                                          -> greater_equals
    | sum ">" sum                                                           -> greater
    | sum "!=" sum                                                          -> not_equals
    | sum "==" sum                                                          -> equals
    | "true"                                                                -> true
    | "false"                                                               -> false

sum: sum "+" term                                                           -> add
    | sum "-" term                                                          -> minus
    | term                                                                  -> term

term: term "*" factor                                                       -> mult
    | term "/" factor                                                       -> div
    | factor                                                                -> factor

factor: "-" factor                                                          -> neg
    | primary                                                               -> primary

primary: SIGNED_NUMBER                                                      -> number
         | WORD                                                             -> variable
         | WORD "[" sum "]"                                                 -> array_indexing
         | "["SIGNED_NUMBER ( "," SIGNED_NUMBER)*"]"                        -> array_indexing_multi
         | "(" sum* ")"                                                     -> grouping

function_call: WORD "(" arg_list? ")"

arg_list: sum ( "," sum )*
